#(3)
def check(a):
    if a>0:
        print('>0')
    elif a < 0:
        print('<0')
    else:
        print("==0")
check(5)

>0

check(-2)

<0

check(0)

#（5）
def avg(a,b):
    return(a+b)/2
    return(a+b) #可以有多条return语句，但只执行一条
avg(4,6)

5.0

#（6）
def avg(a,b=0):#默认参数
    return(a+b)/2
avg(6)

3.0
