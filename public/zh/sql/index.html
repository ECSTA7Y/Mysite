数据合并

横向连接

ALTER TABLE salary
DROP COLUMN Tm;
SELECT * FROM salary
JOIN seasons
ON salary.Player=seasons.Player
  • MySQL多对一、一对多匹配可以自动复制。

纵向连接

  • UNION ALL允许重复值
  • UNION不允许重复值
SELECT Player FROM salary
UNION ALL
SELECT Player FROM seasons; 
ORDER BY Player

  • 在SQL中，视图是基于SQL语句的结果集的可视化的表。
CREATE VIEW teamage AS
SELECT DISTINCT Year,Tm, AVG(Age)
FROM season
GROUP BY Tm
ORDER BY Year;
